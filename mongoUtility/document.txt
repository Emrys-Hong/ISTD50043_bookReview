mongo_book.py
class: MyMongoBook
initialize: mmb = MyMongoBook()
addbook: mmb.addBook('book_metadata','metadata',asin, title=None, price=None, imUrl=None, category=[], salesRank=None, brand=None, also_bought=[], also_viewed=[], buy_after_viewing=[], bought_together=[])
  asin, title, imUrl, brand: string
  price: float
  also_bought, also_viewed, buy_after_viewing, bought_together: list of asin string
  category: list of category name string
  salesRank: python dictionary {category: integer}
  raises exception "Book with given asin already exists!" if asin already exists.
getbook: mmb.getBook('book_metadata','metadata',asin)
  asin: string
  raises exception "Book with given asin is not found!" if no such asin exists.
  raises exceptoin "Found more than one book with the given asin!" if more than one such asin exists.
delbook: mmb.delBook('booke_metadata','metadata',asin)
